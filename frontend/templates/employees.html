async function loadEmployees() { employees = await apiGet('/api/employees'); renderEmployees(employees); } function renderEmployees(list) { const tbody = document.getElementById('empBody'); if (!list.length) { tbody.innerHTML = 'No employees found'; return; } tbody.innerHTML = list.map(e => ` ${e.emp_id} ${e.full_name} ${e.department || '--'} ${e.designation || '--'} ${e.email || '--'} ${e.phone || '--'} ₹${Number(e.salary).toLocaleString('en-IN')} ${e.fingerprint_hash ? 'FP ✓' : ''} ${e.face_encoding ? 'Face ✓' : ''} ${!e.fingerprint_hash && !e.face_encoding ? 'None' : ''} `).join(''); } function filterEmployees() { const q = document.getElementById('searchInp').value.toLowerCase(); renderEmployees(employees.filter(e => e.full_name.toLowerCase().includes(q) || e.emp_id.toLowerCase().includes(q) || (e.department || '').toLowerCase().includes(q) )); } function openModal(emp=null) { editMode = !!emp; document.getElementById('modalTitle').textContent = editMode ? 'Edit Employee' : 'Add Employee'; document.getElementById('fEmpId').disabled = editMode; if (emp) { document.getElementById('fEmpId').value = emp.emp_id; document.getElementById('fName').value = emp.full_name; document.getElementById('fDept').value = emp.department; document.getElementById('fDesig').value = emp.designation; document.getElementById('fEmail').value = emp.email; document.getElementById('fPhone').value = emp.phone; document.getElementById('fSalary').value = emp.salary; document.getElementById('fJoinDate').value = emp.join_date; currentEmpId = emp.emp_id; } else { document.getElementById('empForm').reset(); currentEmpId = null; } document.getElementById('empModal').style.display = 'flex'; } function closeModal() { document.getElementById('empModal').style.display = 'none'; } function editEmployee(empId) { openModal(employees.find(e => e.emp_id === empId)); } async function deleteEmployee(empId) { const emp = employees.find(e => e.emp_id === empId); const name = emp ? emp.full_name : empId; if (!confirm(`Are you sure you want to delete?\n\nEmployee: ${name}\nID: ${empId}`)) return; try { await apiDelete(`/api/employees/${empId}`); // Remove from local array immediately employees = employees.filter(e => e.emp_id !== empId); // Re-render table renderEmployees(employees); // Show green success toast showToast(`✅ ${name} deleted successfully!`, 'success'); } catch(err) { showToast('❌ Error deleting employee. Try again.', 'error'); } } function showToast(message, type) { // Remove existing toast const existing = document.getElementById('toast'); if (existing) existing.remove(); const toast = document.createElement('div'); toast.id = 'toast'; toast.style.cssText = ` position:fixed; top:20px; right:20px; z-index:9999; padding:16px 24px; border-radius:12px; font-size:15px; font-weight:600; box-shadow:0 8px 24px rgba(0,0,0,0.2); display:flex; align-items:center; gap:10px; animation:slideIn 0.3s ease; background:${type === 'success' ? '#22c55e' : '#f72585'}; color:white; max-width:400px; `; toast.innerHTML = message; document.body.appendChild(toast); // Add animation const style = document.createElement('style'); style.textContent = ` @keyframes slideIn { from { transform: translateX(100px); opacity: 0; } to { transform: translateX(0); opacity: 1; } } `; document.head.appendChild(style); // Auto remove after 3 seconds setTimeout(() => { if (toast) toast.remove(); }, 3000); } document.getElementById('empForm').addEventListener('submit', async (e) => { e.preventDefault(); const payload = { emp_id: document.getElementById('fEmpId').value, full_name: document.getElementById('fName').value, department: document.getElementById('fDept').value, designation: document.getElementById('fDesig').value, email: document.getElementById('fEmail').value, phone: document.getElementById('fPhone').value, salary: document.getElementById('fSalary').value, join_date: document.getElementById('fJoinDate').value, }; if (editMode) { await apiPut(`/api/employees/${currentEmpId}`, payload); } else { await apiPost('/api/employees', payload); } closeModal(); await loadEmployees(); showToast(`✅ Employee saved successfully!`, 'success'); }); let bioEmpId = null; function openBioModal(empId, empName) { bioEmpId = empId; document.getElementById('bioEmpName').textContent = `Employee: ${empName} (${empId})`; document.getElementById('bioMsg').textContent = ''; document.getElementById('bioModal').style.display = 'flex'; } async function simulateEnroll(type) { const msgEl = document.getElementById('bioMsg'); msgEl.textContent = `Scanning ${type}... Please wait...`; msgEl.className = 'bio-msg info'; msgEl.style.display = 'block'; await new Promise(r => setTimeout(r, 2000)); const bioData = `${type.toUpperCase()}_${bioEmpId}_${Date.now()}`; try { const res = await apiPost(`/api/employees/${bioEmpId}/enroll`, { type, data: bioData }); if (res.success) { msgEl.textContent = `✅ ${res.message}`; msgEl.className = 'bio-msg success'; // Update employee in local array immediately const emp = employees.find(e => e.emp_id === bioEmpId); if (emp) { if (type === 'fingerprint') { emp.fingerprint_hash = 'enrolled'; } else { emp.face_encoding = 'enrolled'; } } // Re-render table immediately renderEmployees(employees); // Close modal and reload from DB after 2 seconds setTimeout(async () => { document.getElementById('bioModal').style.display = 'none'; await loadEmployees(); // Fresh reload from database }, 2000); } else { msgEl.textContent = '❌ Enrollment failed. Try again.'; msgEl.className = 'bio-msg error'; } } catch(err) { msgEl.textContent = '❌ Server error. Try again.'; msgEl.className = 'bio-msg error'; } } loadEmployees();
